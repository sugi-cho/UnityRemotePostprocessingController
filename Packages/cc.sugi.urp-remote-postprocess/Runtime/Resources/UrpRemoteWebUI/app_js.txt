function closeWebSocket() {
  if (!ws) {
    return;
  }

  try {
    ws.onclose = null;
    ws.close();
  } catch {
  }
  ws = null;
}

function beginGestureLock() {
  gestureLockCount += 1;
  document.body.classList.add("gesture-lock");
}

function endGestureLock() {
  gestureLockCount = Math.max(0, gestureLockCount - 1);
  if (gestureLockCount === 0) {
    document.body.classList.remove("gesture-lock");
  }
}

function bindGestureLock(element) {
  if (!element) {
    return;
  }

  let locked = false;

  const lock = () => {
    if (locked) {
      return;
    }
    locked = true;
    beginGestureLock();
  };

  const unlock = () => {
    if (!locked) {
      return;
    }
    locked = false;
    endGestureLock();
  };

  element.addEventListener("touchstart", lock, { passive: true });
  element.addEventListener("touchend", unlock, { passive: true });
  element.addEventListener("touchcancel", unlock, { passive: true });
  element.addEventListener("pointerdown", (evt) => {
    if (evt.pointerType === "touch") {
      lock();
    }
  }, { passive: true });
  element.addEventListener("pointerup", unlock, { passive: true });
  element.addEventListener("pointercancel", unlock, { passive: true });
  element.addEventListener("lostpointercapture", unlock, { passive: true });
}

window.addEventListener("touchend", () => {

async function api(path, method = "GET", body = null, options = {}) {
  const includeAuth = options.includeAuth !== false;
  const handleUnauthorized = options.handleUnauthorized !== false;
  const headers = {};
  if (body !== null) {
    headers["Content-Type"] = "application/json";
  }
  if (includeAuth && authState.token) {
    headers.Authorization = `Bearer ${authState.token}`;
  }

  const res = await fetch(path, {
    method,
    headers,
    body: body ? JSON.stringify(body) : null
  });

  const raw = await res.text();
  let data = {};
  if (raw) {
    try {
      data = JSON.parse(raw);
    } catch {
      data = {};
    }
  }

  if (!res.ok) {
    const error = new Error(data.error || res.statusText || "request_failed");
    error.status = res.status;
    error.error = data.error || "request_failed";
    error.retryAfterSeconds = Number.isFinite(data.retryAfterSeconds) ? data.retryAfterSeconds : 0;
    if (res.status === 401 && handleUnauthorized) {
      await handleUnauthorizedError(error.error);
    }
    throw error;
  }
  return data;
}

async function refreshAuthStatus() {
  const status = await api("/auth/status", "GET", null, { includeAuth: true, handleUnauthorized: false });
  authState.requiresSetup = !!status.requiresSetup;
  authState.authenticated = !!status.authenticated;

  if (authState.requiresSetup || !authState.authenticated) {
    setAuthToken("");
    closeWebSocket();
  }

  applyAuthUiState();
  return status;
}

async function handleUnauthorizedError() {
  authState.authenticated = false;
  setAuthToken("");
  closeWebSocket();
  stopSyncAutoRefresh();
  await refreshAuthStatus();
  if (authState.requiresSetup) {
    setStatus("Password setup required");
  } else {
    setStatus("Authentication required");
  }
}

function debounce(fn, wait) {
  let timer = null;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), wait);
  };
}

const sendPatchDebounced = debounce(sendPatchNow, 30);

}

function parseJsonValue(raw, fallback) {
  try {
    return JSON.parse(raw);
  } catch {
    return fallback;
  }
}

function toHex(c) {
  return Math.round(Math.max(0, Math.min(1, c)) * 255).toString(16).padStart(2, "0");
}

function hexToRgb(hex) {
  const value = hex.replace("#", "");
  return {
    r: parseInt(value.slice(0, 2), 16) / 255,
    g: parseInt(value.slice(2, 4), 16) / 255,
    b: parseInt(value.slice(4, 6), 16) / 255
  };
}

function hsvToRgb(h, s, v) {
  const i = Math.floor(h * 6);
  const f = h * 6 - i;
  const p = v * (1 - s);
  const q = v * (1 - f * s);
  const t = v * (1 - (1 - f) * s);
  switch (i % 6) {
    case 0: return { r: v, g: t, b: p };
    case 1: return { r: q, g: v, b: p };
    case 2: return { r: p, g: v, b: t };
    case 3: return { r: p, g: q, b: v };
    case 4: return { r: t, g: p, b: v };
    default: return { r: v, g: p, b: q };
  }
}

function rgbToHsv(r, g, b) {
  const max = Math.max(r, g, b);
  const min = Math.min(r, g, b);
  const d = max - min;
  let h = 0;
  const s = max === 0 ? 0 : d / max;
  const v = max;
  if (d !== 0) {
    switch (max) {
      case r: h = ((g - b) / d + (g < b ? 6 : 0)) / 6; break;
      case g: h = ((b - r) / d + 2) / 6; break;
      default: h = ((r - g) / d + 4) / 6; break;
    }
  }
  return { h, s, v };
}

function normalizeRgbFromVector(v) {
  const x = Number.isFinite(v.x) ? Math.max(0, v.x) : 1;
  const y = Number.isFinite(v.y) ? Math.max(0, v.y) : 1;
  const z = Number.isFinite(v.z) ? Math.max(0, v.z) : 1;
  const max = Math.max(x, y, z, 0.0001);
  return { r: x / max, g: y / max, b: z / max, intensity: max };
}

function roundTo3Decimals(value) {
  return Math.round(value * 1000) / 1000;
}

function formatNumericValue(value, allowFloat) {
  if (!allowFloat) {
    return String(Number.parseInt(value, 10) || 0);
  }
  return String(Number(roundTo3Decimals(value).toFixed(3)));
}

function sanitizeNumericText(raw, allowFloat) {
  const source = String(raw ?? "");
  let result = "";
  let hasSign = false;
  let hasDot = false;

  for (let i = 0; i < source.length; i += 1) {
    const ch = source[i];
    if (ch >= "0" && ch <= "9") {
      result += ch;
      continue;
    }
    if (ch === "-" && !hasSign && result.length === 0) {
      hasSign = true;
      result += ch;
      continue;
    }
    if (allowFloat && ch === "." && !hasDot) {
      hasDot = true;
      result += ch;
    }
  }

  return result;
}

function isIntermediateNumericText(text, allowFloat) {
  return text === "" || text === "-" || (allowFloat && (text === "." || text === "-."));
}

function updateRangeVisual(rangeInput) {
  if (!rangeInput) {
    return;
  }

  const min = Number.parseFloat(rangeInput.min);
  const max = Number.parseFloat(rangeInput.max);
  const value = Number.parseFloat(rangeInput.value);
  if (!Number.isFinite(min) || !Number.isFinite(max) || !Number.isFinite(value) || max <= min) {
    rangeInput.style.setProperty("--range-progress", "0%");
    return;
  }

  const progress = Math.max(0, Math.min(100, ((value - min) / (max - min)) * 100));
  rangeInput.style.setProperty("--range-progress", `${progress}%`);
}

function wireRangeVisual(rangeInput) {
  if (!rangeInput || rangeInput.type !== "range") {
    return;
  }

  const refresh = () => updateRangeVisual(rangeInput);
  rangeInput.addEventListener("input", refresh);
  rangeInput.addEventListener("change", refresh);
  refresh();
}

function createFieldFrame(param, options) {
  const wrap = document.createElement("div");
  wrap.className = "field";
  if (!param.overrideState) {
    wrap.classList.add("disabled");
  }

  const head = document.createElement("div");
  head.className = "field-head";
  const name = document.createElement("label");
  name.textContent = options.fieldLabel || param.name;

  const overrideWrap = document.createElement("label");
  overrideWrap.className = "override-wrap";
  const overrideInput = document.createElement("input");
  overrideInput.type = "checkbox";
  overrideInput.checked = !!param.overrideState;
  const overrideText = document.createElement("span");
  overrideText.textContent = "Override";
  overrideWrap.appendChild(overrideInput);
  overrideWrap.appendChild(overrideText);

  head.appendChild(name);
  if (options.showOverrideToggle) {
    head.appendChild(overrideWrap);
  }

  const content = document.createElement("div");
  content.className = "field-content";

  if (options.showOverrideToggle) {
    overrideInput.addEventListener("change", () => {
      param.overrideState = overrideInput.checked;
      wrap.classList.toggle("disabled", !param.overrideState);
      setControlEnabled(content, param.overrideState);
      sendPatchDebounced({
        path: param.path,
        hasOverrideState: true,
        overrideState: param.overrideState,
        valueJson: param.currentValueJson || "null"
      });
    });
  }

  wrap.appendChild(head);
  wrap.appendChild(content);
  return { wrap, content };
}

function setControlEnabled(root, enabled) {
  const controls = root.querySelectorAll("input, select, button");
  controls.forEach((el) => {
    el.disabled = !enabled;
  });
}

function createNumberContent(param, content) {
  const row = document.createElement("div");
  row.className = "row";
  const allowFloat = param.type !== "int";
  const range = document.createElement("input");
  range.type = "range";
  range.min = param.hasMin ? param.min : -10;
  range.max = param.hasMax ? param.max : 10;
  range.step = param.hasStep ? param.step : 0.01;
  const initialRaw = allowFloat
    ? Number.parseFloat(param.currentValueJson || "0")
    : Number.parseInt(param.currentValueJson || "0", 10);
  const initialValue = Number.isFinite(initialRaw)
    ? (allowFloat ? roundTo3Decimals(initialRaw) : initialRaw)
    : 0;
  range.value = String(initialValue);

  const number = document.createElement("input");
  number.type = "text";
  number.className = "small";
  number.inputMode = allowFloat ? "decimal" : "numeric";
  number.pattern = allowFloat ? "-?[0-9]*[.]?[0-9]*" : "-?[0-9]*";
  number.autocomplete = "off";
  number.spellcheck = false;
  number.value = formatNumericValue(initialValue, allowFloat);

  function emit(v) {
    let normalized = allowFloat ? Number.parseFloat(v) : Number.parseInt(v, 10);
    if (!Number.isFinite(normalized)) {
      number.value = String(range.value);
      return;
    }
    if (allowFloat) {
      normalized = roundTo3Decimals(normalized);
    }
    range.value = String(normalized);
    updateRangeVisual(range);
    number.value = formatNumericValue(normalized, allowFloat);
    param.currentValueJson = number.value;
    sendPatchDebounced({
      path: param.path,
      hasOverrideState: true,
      overrideState: !!param.overrideState,
      valueJson: param.currentValueJson
    });
  }

  function commitNumberInput() {
    const filtered = sanitizeNumericText(number.value, allowFloat);
    number.value = filtered;
    if (isIntermediateNumericText(filtered, allowFloat)) {
      number.value = String(range.value);
      return;
    }
    emit(filtered);
  }

  wireRangeVisual(range);
  range.addEventListener("input", () => emit(range.value));
  number.addEventListener("input", () => {
    const filtered = sanitizeNumericText(number.value, allowFloat);
    if (filtered !== number.value) {
      number.value = filtered;
    }
  });
  number.addEventListener("change", commitNumberInput);
  number.addEventListener("blur", commitNumberInput);
  number.addEventListener("focus", () => {
    number.select();
  });
  number.addEventListener("pointerup", () => {
    if (document.activeElement === number) {
      number.select();
    }
  });
  bindGestureLock(range);
  row.appendChild(range);
  row.appendChild(number);
  content.appendChild(row);
}

function createColorContent(param, content) {
  const row = document.createElement("div");
  row.className = "row";
  const picker = document.createElement("input");
  picker.type = "color";
  const alpha = document.createElement("input");
  alpha.type = "range";
  alpha.min = 0;
  alpha.max = 1;
  alpha.step = 0.01;
  const preview = document.createElement("div");
  preview.className = "color-preview";

  const value = parseJsonValue(param.currentValueJson, { r: 1, g: 1, b: 1, a: 1 });
  picker.value = `#${toHex(value.r)}${toHex(value.g)}${toHex(value.b)}`;
  alpha.value = Number.isFinite(value.a) ? value.a : 1;
  wireRangeVisual(alpha);

  function emit() {
    const rgb = hexToRgb(picker.value);
    const rgba = { r: rgb.r, g: rgb.g, b: rgb.b, a: Number.parseFloat(alpha.value) };
    preview.style.background = `rgba(${Math.round(rgba.r * 255)}, ${Math.round(rgba.g * 255)}, ${Math.round(rgba.b * 255)}, ${rgba.a})`;
    param.currentValueJson = JSON.stringify(rgba);
    sendPatchDebounced({
      path: param.path,
      hasOverrideState: true,
      overrideState: !!param.overrideState,
      valueJson: param.currentValueJson
    });
  }

  picker.addEventListener("input", emit);
  alpha.addEventListener("input", emit);
  bindGestureLock(picker);
  bindGestureLock(alpha);
  emit();
  row.appendChild(picker);
  row.appendChild(alpha);
  row.appendChild(preview);
  content.appendChild(row);
}

function createVectorContent(param, content, dimensions) {
  const row = document.createElement("div");
  row.className = "row";
  const current = parseJsonValue(param.currentValueJson, {});
  const keys = ["x", "y", "z", "w"].slice(0, dimensions);
  const inputs = {};

  keys.forEach((k) => {
    const input = document.createElement("input");
    input.type = "number";
    input.step = "0.01";
    input.className = "small";
    input.value = Number.isFinite(current[k]) ? current[k] : 0;
    row.appendChild(input);
    inputs[k] = input;
  });

  function emit() {
    const payload = {};
    keys.forEach((k) => {
      payload[k] = Number.parseFloat(inputs[k].value) || 0;
    });
    param.currentValueJson = JSON.stringify(payload);
    sendPatchDebounced({
      path: param.path,
      hasOverrideState: true,
      overrideState: !!param.overrideState,
      valueJson: param.currentValueJson
    });
  }

  keys.forEach((k) => inputs[k].addEventListener("change", emit));
  content.appendChild(row);
}

function createTrackballLikeContent(param, content) {
  const current = parseJsonValue(param.currentValueJson, { x: 1, y: 1, z: 1, w: 0 });
  const normalized = normalizeRgbFromVector(current);
  const hsv = rgbToHsv(normalized.r, normalized.g, normalized.b);
  const theta0 = hsv.h * Math.PI * 2;
  let wheelVec = {
    x: Math.cos(theta0 + Math.PI * 0.5) * hsv.s,
    y: Math.sin(theta0 - Math.PI * 0.5) * hsv.s
  };

  const container = document.createElement("div");
  container.className = "trackball";
  const wheel = document.createElement("div");
  wheel.className = "trackball-wheel";
  const picker = document.createElement("input");
  picker.type = "color";
  picker.className = "trackball-picker";
  picker.value = `#${toHex(normalized.r)}${toHex(normalized.g)}${toHex(normalized.b)}`;
  const marker = document.createElement("div");
  marker.className = "trackball-marker";
  wheel.appendChild(picker);
  wheel.appendChild(marker);

  const sliders = document.createElement("div");
  sliders.className = "trackball-sliders";

  const intensityLabel = document.createElement("label");
  intensityLabel.textContent = "Balance";
  const intensity = document.createElement("input");
  intensity.type = "range";
  intensity.min = "0";
  intensity.max = "2";
  intensity.step = "0.01";
  intensity.value = String(Math.max(0, Math.min(2, normalized.intensity)));

  const offsetLabel = document.createElement("label");
  offsetLabel.textContent = "Offset";
  const offset = document.createElement("input");
  offset.type = "range";
  offset.min = "-1";
  offset.max = "1";
  offset.step = "0.01";
  offset.value = String(Number.isFinite(current.w) ? current.w : 0);
  wireRangeVisual(intensity);
  wireRangeVisual(offset);

  const valueEditors = document.createElement("div");
  valueEditors.className = "trackball-values";
  const rgbwInputs = {};
  const labelByKey = { x: "R", y: "G", z: "B", w: "W" };
  ["x", "y", "z", "w"].forEach((key) => {
    const item = document.createElement("label");
    item.className = "trackball-value-item";
    const label = document.createElement("span");
    label.textContent = labelByKey[key] || key.toUpperCase();
    const input = document.createElement("input");
    input.type = "text";
    input.className = "trackball-value-input";
    input.inputMode = "decimal";
    input.pattern = "-?[0-9]*[.]?[0-9]*";
    input.autocomplete = "off";
    input.spellcheck = false;
    input.addEventListener("input", () => {
      const filtered = sanitizeNumericText(input.value, true);
      if (filtered !== input.value) {
        input.value = filtered;
      }
    });
    input.addEventListener("focus", () => input.select());
    input.addEventListener("pointerup", () => {
      if (document.activeElement === input) {
        input.select();
      }
    });
    item.appendChild(label);
    item.appendChild(input);
    valueEditors.appendChild(item);
    rgbwInputs[key] = input;
  });
  const maxMarkerRadius = 42;

  function setMarkerFromWheelVec() {
    marker.style.left = `${60 + wheelVec.x * maxMarkerRadius}px`;
    marker.style.top = `${60 + wheelVec.y * maxMarkerRadius}px`;
  }

  function setRgbwInputs(payload) {
    rgbwInputs.x.value = formatNumericValue(payload.x, true);
    rgbwInputs.y.value = formatNumericValue(payload.y, true);
    rgbwInputs.z.value = formatNumericValue(payload.z, true);
    rgbwInputs.w.value = formatNumericValue(payload.w, true);
  }

  function parseRgbwInputValue(key, fallback) {
    const input = rgbwInputs[key];
    const filtered = sanitizeNumericText(input.value, true);
    input.value = filtered;
    if (isIntermediateNumericText(filtered, true)) {
      return fallback;
    }
    const parsed = Number.parseFloat(filtered);
    return Number.isFinite(parsed) ? roundTo3Decimals(parsed) : fallback;
  }

  function applyPayloadFromRgbw(payload) {
    const x = Number.isFinite(payload.x) ? payload.x : 0;
    const y = Number.isFinite(payload.y) ? payload.y : 0;
    const z = Number.isFinite(payload.z) ? payload.z : 0;
    const w = Number.isFinite(payload.w) ? payload.w : 0;
    const normalizedRgb = normalizeRgbFromVector({ x, y, z });
    const hsvFromRgb = rgbToHsv(normalizedRgb.r, normalizedRgb.g, normalizedRgb.b);
    const theta = hsvFromRgb.h * Math.PI * 2;
    wheelVec = {
      x: Math.cos(theta + Math.PI * 0.5) * hsvFromRgb.s,
      y: Math.sin(theta - Math.PI * 0.5) * hsvFromRgb.s
    };
    intensity.value = String(Math.max(0, Math.min(2, roundTo3Decimals(normalizedRgb.intensity))));
    offset.value = String(Math.max(-1, Math.min(1, roundTo3Decimals(w))));
    updateRangeVisual(intensity);
    updateRangeVisual(offset);
    updatePreview();
  }

  function commitRgbwInputs() {
    const currentPayload = parseJsonValue(param.currentValueJson, { x: 0, y: 0, z: 0, w: 0 });
    applyPayloadFromRgbw({
      x: parseRgbwInputValue("x", Number.isFinite(currentPayload.x) ? currentPayload.x : 0),
      y: parseRgbwInputValue("y", Number.isFinite(currentPayload.y) ? currentPayload.y : 0),
      z: parseRgbwInputValue("z", Number.isFinite(currentPayload.z) ? currentPayload.z : 0),
      w: parseRgbwInputValue("w", Number.isFinite(currentPayload.w) ? currentPayload.w : 0)
    });
  }

  function updatePreview() {
    const dx = wheelVec.x;
    const dy = wheelVec.y;
    const sat = Math.max(0, Math.min(1, Math.hypot(dx, dy)));
    const angle = Math.atan2(dx, -dy);
    const hue = 1 - ((angle > 0 ? angle : (Math.PI * 2) + angle) / (Math.PI * 2));
    const rgb = hsvToRgb(hue, sat, 1);
    picker.value = `#${toHex(rgb.r)}${toHex(rgb.g)}${toHex(rgb.b)}`;
    const i = Number.parseFloat(intensity.value) || 0;
    const w = Number.parseFloat(offset.value) || 0;
    const payload = {
      x: roundTo3Decimals(rgb.r * i),
      y: roundTo3Decimals(rgb.g * i),
      z: roundTo3Decimals(rgb.b * i),
      w: roundTo3Decimals(w)
    };
    param.currentValueJson = JSON.stringify(payload);
    marker.style.background = picker.value;
    setMarkerFromWheelVec();
    setRgbwInputs(payload);
    sendPatchDebounced({
      path: param.path,
      hasOverrideState: true,
      overrideState: !!param.overrideState,
      valueJson: param.currentValueJson
    });
  }

  function updateWheelVecFromPointer(evt) {
    const rect = wheel.getBoundingClientRect();
    const cx = rect.left + rect.width * 0.5;
    const cy = rect.top + rect.height * 0.5;
    let dx = (evt.clientX - cx) / maxMarkerRadius;
    let dy = (evt.clientY - cy) / maxMarkerRadius;
    const len = Math.hypot(dx, dy);
    if (len > 1) {
      dx /= len;
      dy /= len;
    }
    wheelVec = { x: dx, y: dy };
    updatePreview();
  }

  let dragging = false;
  wheel.addEventListener("pointerdown", (evt) => {
    dragging = true;
    wheel.setPointerCapture(evt.pointerId);
    updateWheelVecFromPointer(evt);
  });
  wheel.addEventListener("pointermove", (evt) => {
    if (!dragging) return;
    updateWheelVecFromPointer(evt);
  });
  wheel.addEventListener("pointerup", (evt) => {
    dragging = false;
    wheel.releasePointerCapture(evt.pointerId);
  });
  wheel.addEventListener("pointercancel", () => {
    dragging = false;
  });
  bindGestureLock(wheel);

  picker.addEventListener("input", () => {
    const rgb = hexToRgb(picker.value);
    const h = rgbToHsv(rgb.r, rgb.g, rgb.b);
    const theta = h.h * Math.PI * 2;
    wheelVec = {
      x: Math.cos(theta + Math.PI * 0.5) * h.s,
      y: Math.sin(theta - Math.PI * 0.5) * h.s
    };
    updatePreview();
  });
  intensity.addEventListener("input", updatePreview);
  offset.addEventListener("input", updatePreview);
  bindGestureLock(intensity);
  bindGestureLock(offset);
  ["x", "y", "z", "w"].forEach((key) => {
    rgbwInputs[key].addEventListener("change", commitRgbwInputs);
    rgbwInputs[key].addEventListener("blur", commitRgbwInputs);
  });

  sliders.appendChild(intensityLabel);
  sliders.appendChild(intensity);
  sliders.appendChild(offsetLabel);
  sliders.appendChild(offset);
  sliders.appendChild(valueEditors);

  container.appendChild(wheel);
  container.appendChild(sliders);
  content.appendChild(container);
  setMarkerFromWheelVec();
  updatePreview();
}

function createEnumContent(param, content) {
  const row = document.createElement("div");
  row.className = "row";
  const select = document.createElement("select");
  const current = Number.parseInt(param.currentValueJson || "0", 10);
  const optionsContainer = parseJsonValue(param.enumOptionsJson, { options: [] });
  (optionsContainer.options || []).forEach((opt) => {
    const option = document.createElement("option");
    option.value = String(opt.value);
    option.textContent = opt.label;
    if (opt.value === current) {
      option.selected = true;
    }
    select.appendChild(option);
  });

  select.addEventListener("change", () => {
    param.currentValueJson = String(Number.parseInt(select.value, 10) || 0);
    sendPatchDebounced({
      path: param.path,
      hasOverrideState: true,
      overrideState: !!param.overrideState,
      valueJson: param.currentValueJson
    });
  });

  row.appendChild(select);
  content.appendChild(row);
}

function createField(param, options) {
  const { wrap, content } = createFieldFrame(param, options);
  const componentType = options.componentTypeName || "";
  const isTrackballTarget =
    param.type === "vector4" &&
    (componentType === "LiftGammaGain" || componentType === "ShadowsMidtonesHighlights");

  if (isTrackballTarget) createTrackballLikeContent(param, content);
  else if (param.type === "bool") createToggleContent(param, content);
  else if (param.type === "int" || param.type === "float") createNumberContent(param, content);
  else if (param.type === "color") createColorContent(param, content);
  else if (param.type === "vector2") createVectorContent(param, content, 2);
  else if (param.type === "vector3") createVectorContent(param, content, 3);
  else if (param.type === "vector4") createVectorContent(param, content, 4);
  else if (param.type === "enum") createEnumContent(param, content);
  else {
    const unsupported = document.createElement("div");
    unsupported.textContent = `${param.type} is not supported`;
    content.appendChild(unsupported);
  }

  const enabled = options.showOverrideToggle ? !!param.overrideState : true;
  setControlEnabled(content, enabled);
  return wrap;
}

function filterTonemappingParameters(component, sourceParams) {
  if ((component.typeName || "") !== "Tonemapping") {
    return sourceParams;
  }

  // Inspector寄せ: Tonemappingは常にModeを優先表示
  const modeParam = sourceParams.find((p) => p.path.endsWith("/mode"));
  if (!modeParam) {
    return sourceParams;
  }

  return [modeParam];
}

function createComponentNode(component, options) {
  if (options.onlyOverrideOn && !component.overrideState) {
    return null;
  }

  const filtered = (component.parameters || []).filter((p) => {
    return options.onlyOverrideOn ? !!p.overrideState : true;
  });
  const parameters = filterTonemappingParameters(component, filtered);
  if (parameters.length === 0) {
    return null;
  }

  const details = document.createElement("details");
  details.className = "component";
  details.open = true;

  const summary = document.createElement("summary");
  summary.className = "component-header";
  const title = document.createElement("span");
  title.textContent = component.displayName || component.typeName;
  summary.appendChild(title);

  if (options.showOverrideToggle) {
    const componentOverrideWrap = document.createElement("label");
    componentOverrideWrap.className = "override-wrap component-override";
    const componentOverrideInput = document.createElement("input");
    componentOverrideInput.type = "checkbox";
    componentOverrideInput.checked = !!component.overrideState;
    const componentOverrideText = document.createElement("span");
    componentOverrideText.textContent = "Component Override";
    componentOverrideWrap.appendChild(componentOverrideInput);
    componentOverrideWrap.appendChild(componentOverrideText);

    componentOverrideWrap.addEventListener("click", (evt) => {
      evt.stopPropagation();
    });

    componentOverrideInput.addEventListener("change", () => {
      component.overrideState = componentOverrideInput.checked;
      setControlEnabled(body, component.overrideState);
      sendPatchDebounced({
        path: component.componentPath,
        hasOverrideState: true,
        overrideState: component.overrideState,
        valueJson: "null"
      });
    });

    summary.appendChild(componentOverrideWrap);
  }
  details.appendChild(summary);

  const body = document.createElement("div");
  body.className = "component-body";
  if (component.typeName === "ChannelMixer") {
    renderChannelMixerFields(component, parameters, body, options);
  } else {
    parameters.forEach((param) => {
      body.appendChild(createField(param, { ...options, componentTypeName: component.typeName }));
    });
  }
  if (options.showOverrideToggle) {
    setControlEnabled(body, !!component.overrideState);
  }

  details.appendChild(body);
  return details;
}

function renderChannelMixerFields(component, parameters, body, options) {
  const groups = {
    red: { title: "Red", params: [] },
    green: { title: "Green", params: [] },
    blue: { title: "Blue", params: [] }
  };

  parameters.forEach((param) => {
    if (param.path.endsWith("/redOutRedIn")) groups.red.params.push({ param, label: "Red" });
    if (param.path.endsWith("/redOutGreenIn")) groups.red.params.push({ param, label: "Green" });
    if (param.path.endsWith("/redOutBlueIn")) groups.red.params.push({ param, label: "Blue" });
    if (param.path.endsWith("/greenOutRedIn")) groups.green.params.push({ param, label: "Red" });
    if (param.path.endsWith("/greenOutGreenIn")) groups.green.params.push({ param, label: "Green" });
    if (param.path.endsWith("/greenOutBlueIn")) groups.green.params.push({ param, label: "Blue" });
    if (param.path.endsWith("/blueOutRedIn")) groups.blue.params.push({ param, label: "Red" });
    if (param.path.endsWith("/blueOutGreenIn")) groups.blue.params.push({ param, label: "Green" });
    if (param.path.endsWith("/blueOutBlueIn")) groups.blue.params.push({ param, label: "Blue" });
  });

  const tabKeys = ["red", "green", "blue"].filter((k) => groups[k].params.length > 0);
  if (tabKeys.length === 0) {
    return;
  }

  let selected = channelMixerTabByComponent[component.componentPath] || tabKeys[0];
  if (!tabKeys.includes(selected)) {
    selected = tabKeys[0];
  }
  channelMixerTabByComponent[component.componentPath] = selected;

  const tabRow = document.createElement("div");
  tabRow.className = "channel-tabs";
  tabKeys.forEach((key) => {
    const btn = document.createElement("button");
    btn.type = "button";
    btn.textContent = groups[key].title;
    btn.className = key === selected ? "is-active" : "";
    btn.addEventListener("click", () => {
      channelMixerTabByComponent[component.componentPath] = key;
      renderSchema(state.schema);
    });
    tabRow.appendChild(btn);
  });
  body.appendChild(tabRow);

  groups[selected].params.forEach((item) => {
    body.appendChild(createField(item.param, {
      ...options,
      componentTypeName: component.typeName,
      fieldLabel: item.label
    }));
  });
}

function renderSchema(schema) {
  appEl.innerHTML = "";
  const options = {
    onlyOverrideOn: currentView === "active",
    showOverrideToggle: currentView === "all"
  };

  (schema.volumes || []).forEach((volume) => {
    const visibleComponents = [];
    (volume.components || []).forEach((component) => {
      const node = createComponentNode(component, options);
      if (node) {
        visibleComponents.push(node);
      }
    });
    if (visibleComponents.length === 0) {
      return;
    }

    const volumeDetails = document.createElement("details");
    volumeDetails.className = "volume";
    volumeDetails.open = true;

    const volumeSummary = document.createElement("summary");
    volumeSummary.className = "volume-header";
    volumeSummary.textContent = volume.displayName || volume.volumeId;
    volumeDetails.appendChild(volumeSummary);

    const volumeBody = document.createElement("div");
    volumeBody.className = "volume-body";
    visibleComponents.forEach((node) => {
      volumeBody.appendChild(node);
    });

    volumeDetails.appendChild(volumeBody);
    appEl.appendChild(volumeDetails);
  });
}

function mergeCurrentState(schema, patch) {
  const valueMap = {};
  const overrideMap = {};
  (patch.entries || []).forEach((e) => {
    valueMap[e.path] = e.valueJson;
    if (e.hasOverrideState) {
      overrideMap[e.path] = e.overrideState;
    }
  });

  (schema.volumes || []).forEach((v) => {
    (v.components || []).forEach((c) => {
      if (overrideMap[c.componentPath] !== undefined) {
        c.overrideState = overrideMap[c.componentPath];
      }
      (c.parameters || []).forEach((p) => {
        if (valueMap[p.path] !== undefined) p.currentValueJson = valueMap[p.path];
        if (overrideMap[p.path] !== undefined) p.overrideState = overrideMap[p.path];
      });
    });
  });
}

async function refreshSchema() {
  const schema = await api("/schema");
  const liveState = await api("/state");
  mergeCurrentState(schema, liveState);
  state.schema = schema;
  renderSchema(schema);
}

function formatUtcToLocal(utcText) {
  if (!utcText) {
    return "-";
  }
  const date = new Date(utcText);
  if (Number.isNaN(date.getTime())) {
    return "-";
  }
  return date.toLocaleString();
}

function renderSyncTargets() {
  if (!syncTargetsEl) {
    return;
  }

  syncTargetsEl.innerHTML = "";
  const targets = (syncTargetsState && syncTargetsState.targets) || [];
  if (targets.length === 0) {
    const empty = document.createElement("div");
    empty.className = "sync-target-empty";
    empty.textContent = "No sync targets configured";
    syncTargetsEl.appendChild(empty);
    return;
  }

  targets.forEach((target) => {
    const item = document.createElement("div");
    item.className = "sync-target-item";

    const head = document.createElement("div");
    head.className = "sync-target-head";
    const address = document.createElement("div");
    address.className = "sync-target-address";
    address.textContent = target.address || "(unknown)";
    const badge = document.createElement("span");
    badge.className = "sync-target-status";

    let statusLabel = target.connected ? "Connected" : "Disconnected";
    if (!target.enabled) {
      statusLabel = "Disabled";
      badge.classList.add("is-disabled");
    } else if (target.connected) {
      badge.classList.add("is-connected");
    }
    badge.textContent = statusLabel;
    head.appendChild(address);
    head.appendChild(badge);

    const meta = document.createElement("div");
    meta.className = "sync-target-meta";
    const checkedAt = formatUtcToLocal(target.lastCheckedUtc);
    const syncedAt = formatUtcToLocal(target.lastSyncedUtc);
    const lastError = (target.lastError || "").trim();
    meta.textContent = `Last check: ${checkedAt} / Last sync: ${syncedAt}${lastError ? ` / Error: ${lastError}` : ""}`;

    const actions = document.createElement("div");
    actions.className = "sync-target-actions";
    const toggleBtn = document.createElement("button");
    toggleBtn.type = "button";
    toggleBtn.textContent = target.enabled ? "Disable" : "Enable";
    toggleBtn.addEventListener("click", async () => {
      try {
        await api("/sync/targets/set-enabled", "POST", { targetId: target.id, enabled: !target.enabled });
        await refreshSyncTargets({ silent: true });
        setStatus(target.enabled ? "Sync target disabled" : "Sync target enabled");
      } catch (e) {
        setStatus(`Sync toggle error: ${e.message}`);
      }
    });

    const removeBtn = document.createElement("button");
    removeBtn.type = "button";
    removeBtn.className = "danger";
    removeBtn.textContent = "Remove";
    removeBtn.addEventListener("click", async () => {
      if (!confirm(`Remove sync target "${target.address}"?`)) {
        return;
      }

      try {
        await api("/sync/targets/remove", "POST", { targetId: target.id });
        await refreshSyncTargets({ silent: true });
        setStatus("Sync target removed");
      } catch (e) {
        setStatus(`Sync remove error: ${e.message}`);
      }
    });

    actions.appendChild(toggleBtn);
    actions.appendChild(removeBtn);
    item.appendChild(head);
    item.appendChild(meta);
    item.appendChild(actions);
    syncTargetsEl.appendChild(item);
  });
}

async function refreshSyncTargets(options = {}) {
  if (!authState.authenticated) {
    syncTargetsState = { targets: [] };
    if (currentView === "sync") {
      renderSyncTargets();
    }
    return;
  }

  const silent = !!options.silent;
  try {
    const data = await api("/sync/targets");
    syncTargetsState = data || { targets: [] };
    if (currentView === "sync") {
      renderSyncTargets();
    }
  } catch (e) {
    if (!silent) {
      setStatus(`Sync list error: ${e.message}`);
      throw e;
    }
  }
}

function stopSyncAutoRefresh() {
  if (!syncRefreshTimer) {
    return;
  }

  clearInterval(syncRefreshTimer);
  syncRefreshTimer = null;
}

function startSyncAutoRefresh() {
  stopSyncAutoRefresh();
  syncRefreshTimer = setInterval(() => {
    if (!authState.authenticated || currentView !== "sync") {
      return;
    }

    refreshSyncTargets({ silent: true }).catch(() => {});
  }, 3000);
}

function connectWebSocket() {
  if (!authState.authenticated || !authState.token) {
    return;
  }

  const protocol = location.protocol === "https:" ? "wss" : "ws";
  const tokenQuery = `token=${encodeURIComponent(authState.token)}`;
  const url = `${protocol}://${location.host}/ws?${tokenQuery}`;
  ws = new WebSocket(url);
  ws.onopen = () => setStatus("Connected");
  ws.onclose = async () => {
    ws = null;
    if (!authState.authenticated) {
      return;
    }

    setStatus("Disconnected. reconnecting...");
    setTimeout(async () => {
      if (!authState.authenticated) {
        return;
      }

      try {
        await refreshAuthStatus();
        if (!authState.authenticated) {
          setStatus(authState.requiresSetup ? "Password setup required" : "Authentication required");
          return;
        }
        connectWebSocket();
      } catch (e) {
        setStatus(`Reconnect error: ${e.message}`);
      }
    }, 1200);
  };
  ws.onerror = () => {
    if (authState.authenticated) {
      setStatus("WebSocket error");
    }
  };
  ws.onmessage = async (evt) => {
    try {
      const message = JSON.parse(evt.data);
      if (message.eventName === "state.updated" || message.eventName === "preset.loaded") {
        await refreshSchema();
      }
      if (message.eventName === "error" && message.message) {
        setStatus(`Server error: ${message.message}`);
      }
    } catch {
    }
  };
}

async function startAuthorizedSession() {
  await refreshPresetList();
  await refreshSchema();
  await refreshSyncTargets({ silent: true });
  setView(currentView);
  startSyncAutoRefresh();
  closeWebSocket();
  connectWebSocket();
  setStatus("Connected");
}

async function setupPassword() {
  const password = authPasswordEl.value || "";
  const confirmPassword = authPasswordConfirmEl.value || "";
  if (password.length < 8) {
    setStatus("Password must be at least 8 characters");
    return;
  }
  if (password !== confirmPassword) {
    setStatus("Password confirmation does not match");
    return;
  }

  try {
    const result = await api("/auth/setup", "POST", { password, confirmPassword }, { includeAuth: false, handleUnauthorized: false });
    setAuthToken(result.token || "");
    authState.requiresSetup = false;
    authState.authenticated = !!result.token;
    applyAuthUiState();
    await startAuthorizedSession();
  } catch (e) {
    setStatus(`Setup error: ${e.message}`);
  }
}

async function login() {
  const password = authPasswordEl.value || "";
  if (!password) {
    setStatus("Please enter password");
    return;
  }

  try {
    const result = await api("/auth/login", "POST", { password }, { includeAuth: false, handleUnauthorized: false });
    setAuthToken(result.token || "");
    authState.requiresSetup = false;
    authState.authenticated = !!result.token;
    applyAuthUiState();
    await startAuthorizedSession();
  } catch (e) {
    if (e.status === 429 && e.retryAfterSeconds > 0) {
      setStatus(`Login temporarily locked. retry after ${e.retryAfterSeconds}s`);
      return;
    }
    setStatus(`Login error: ${e.message}`);
  }
}

async function logout() {
  try {
    await api("/auth/logout", "POST", null, { includeAuth: true, handleUnauthorized: false });
  } catch {
  }

  authState.authenticated = false;
  setAuthToken("");
  closeWebSocket();
  stopSyncAutoRefresh();
  await refreshAuthStatus();
  setStatus(authState.requiresSetup ? "Password setup required" : "Logged out");
}

async function main() {

    setView(currentView);

      return;
    }

    setStatus(authState.requiresSetup ? "Password setup required" : "Please login");
  } catch (e) {
    setStatus(`Connection error: ${e.message}`);
  }
}

main();
